function calcular() {

  let precioM2 = parseFloat(document.getElementById("precioM2").value) || 0;

  let largo = parseFloat(document.getElementById("largo").value) || 0;
  let ancho = parseFloat(document.getElementById("ancho").value) || 0;

  let zocaloML = parseFloat(document.getElementById("zocalo").value) || 0;
  let regruesoML = parseFloat(document.getElementById("regrueso").value) || 0;

  let grif = parseInt(document.getElementById("grif").value) || 0;
  let bacha = parseInt(document.getElementById("bacha").value) || 0;
  let anafe = parseInt(document.getElementById("anafe").value) || 0;

  // PRECIOS FIJOS (después los ajustás)
  let precioGrif = 50;
  let precioBacha = 80;
  let precioAnafe = 100;

  let m2 = largo * ancho;
  let base = m2 * precioM2;

  let zocalo = zocaloML * precioM2 * 0.10;
  let regrueso = regruesoML * precioM2 * 0.20;

  let traforos = (grif * precioGrif) + (bacha * precioBacha) + (anafe * precioAnafe);

  let subtotal = base + zocalo + regrueso + traforos;

  let flete = Math.max(subtotal * 0.15, 400);

  let total = subtotal + flete;

  document.getElementById("resultado").innerText =
    "Total: $" + total.toFixed(2);
}
